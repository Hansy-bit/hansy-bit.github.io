document.querySelectorAll('img.expandable').forEach(function (img) {
  function expand() {
    var box = img.parentElement.clientWidth; // square side in px
    var nw = img.naturalWidth, nh = img.naturalHeight;
    if (!nw || !nh) return;
    var ratio = nw / nh;
    // Keep the side that already fits at the box size; grow only the cropped side.
    if (ratio >= 1) {            // landscape: height fits, width was cropped
      img.style.height = box + 'px';
      img.style.width = (box * ratio) + 'px';
    } else {                     // portrait: width fits, height was cropped
      img.style.width = box + 'px';
      img.style.height = (box / ratio) + 'px';
    }
    img.classList.add('expanded');
  }
  function reset() {
    img.style.width = '';
    img.style.height = '';
    img.classList.remove('expanded');
  }
  img.addEventListener('mouseenter', expand);
  img.addEventListener('mouseleave', reset);
});
</script>
